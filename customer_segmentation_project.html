05 — Code Sample

K-Means implementation

Core clustering logic using scikit-learn — finding the optimal number of segments via the Elbow Method.

import pandas as pd
import numpy as np
from sklearn.cluster import KMeans
from sklearn.preprocessing import StandardScaler
import matplotlib.pyplot as plt

# Load and prepare customer data
df = pd.read_csv('customer_data.csv')
features = ['purchase_frequency', 'avg_order_value',
             'recency_days', 'total_spend', 'age']

# Normalize features
scaler = StandardScaler()
X_scaled = scaler.fit_transform(df[features])

# Elbow Method — find optimal k
inertias = []
for k in range(1, 11):
    km = KMeans(n_clusters=k, random_state=42, n_init=10)
    km.fit(X_scaled)
    inertias.append(km.inertia_)

# Fit final model with k=4 segments
kmeans = KMeans(n_clusters=4, random_state=42, n_init=10)
df['Segment'] = kmeans.fit_predict(X_scaled)

# Analyze each segment
segment_profile = df.groupby('Segment')[features].mean().round(2)
print(segment_profile)
